00103         #  @param in_criterias This dictionary contains selections' criterions.
00104         #         <ul>
00105         #             <li>Dictionary's key : the name of a column.</li>
00106         #             <li>Dictionary's value: this value can be : a simple value, a regular expression or a functio

00110         #  @param in_action This parameter defines the selection's method.

00114         #                 Dictionary's values must be simple values.</li>

00116         #                 Dictionary's values must be regular expressions.</li>
00117         #             <li>Csv.EXECUTE (3): Execution of code.
00118         #                 Dictionary's values must be functions.
00119         #                 Function's signature is: def myFuntion(in_value)</li>
00120         #         </ul>
00121         # @return The method returns a lust of records.
00122         def select(self, in_criterias=None, in_action=1):
00123                 if in_criterias is None: return self.__records;
00124                 res = []
00125                 for record in self.__records:
00126                         r = True
00127                         for key, value in in_criterias.items():
00128                                 if not key in self.__header: raise RuntimeError('Unexpected name (%s) for criteria!' % key)
00129                                 record_value = record[self.__position(key)]
00130                                 
00131                                 # Simple comparaison.
00132                                 if in_action == Csv.EQUALITY:
00133                                         if not value == record_value:
00134                                                 r = False
00135                                                 break
00136                                 else:
00137                                         # Matching.
00138                                         if in_action == Csv.MATCH:
00139                                                 reg = re.compile(value)
00140                                                 if not re.match(reg, record_value):
00141                                                         r = False
00142                                                         break
00143                                         else:
00144                                                 # Verify through function's execution.
00145                                                 if in_action == Csv.EXECUTE:
00146                                                         if not value(record_value):
00147                                                                 r = False
00148                                                                 break
00149                                                 else:
00150                                                         raise RuntimeError('Invalid select method (%d)!' % in_action)
00151                         if r: res.append(record)
00152                 return res
