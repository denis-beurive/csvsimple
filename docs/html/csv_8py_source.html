00132         #  @param in_criterias This dictionary contains selections' criterions.
00133         #         <ul>
00134         #             <li>Dictionary's key : the name of a column.</li>
00135         #             <li>Dictionary's value: this value can be : a simple value, a regular expression or a functio
00136         #                 The type of value depends on the value of parameter "in_action".</li>
00137         #         </ul>
00138         #         If this parameter is not specified, then the method returns all re records in the container.
00139         #  @param in_action This parameter defines the selection's method. 
00140         #         Three methods are available :
00141         #         <ul>
00142         #             <li>Csv.EQUALITY (1): Simple equality.
00143         #                 Dictionary's values must be simple values.</li>
00144         #             <li>Csv.MATCH (2): Pattern matching.
00145         #                 Dictionary's values must be regular expressions.</li>
00146         #             <li>Csv.EXECUTE (3): Execution of code.
00147         #                 Dictionary's values must be functions.
00148         #                 Function's signature is: def myFuntion(in_value)</li>
00149         #         </ul>
00150         # @return The method returns a lust of records.
00151         def select(self, in_criterias=None, in_action=1):
00152                 if in_criterias is None: return self.__records;
00153                 res = []
00154                 for record in self.__records:
00155                         r = True
00156                         for key, value in in_criterias.items():
00157                                 if not key in self.__header: raise RuntimeError('Unexpected name (%s) for criteria!' % key)
00158                                 record_value = record[self.__position(key)]
00159                                 
00160                                 # Simple comparaison.
00161                                 if in_action == Csv.EQUALITY:
00162                                         if not value == record_value:
00163                                                 r = False
00164                                                 break
00165                                 else:
00166                                         # Matching.
00167                                         if in_action == Csv.MATCH:
00168                                                 reg = re.compile(value)
00169                                                 if not re.match(reg, record_value):
00170                                                         r = False
00171                                                         break
00172                                         else:
00173                                                 # Verify through function's execution.
00174                                                 if in_action == Csv.EXECUTE:
00175                                                         if not value(record_value):
00176                                                                 r = False
00177                                                                 break
00178                                                 else:
00179                                                         raise RuntimeError('Invalid select method (%d)!' % in_action)
00180                         if r: res.append(record)
00181                 return res
